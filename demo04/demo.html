点击每个模块查看输出；

1、逻辑或操作（短路操作）： 
        function orRight() {
            var rightValue = true;
            alert(rightValue || falseValue);
        }
2、逻辑或操作（报错）： 
        function orFalse() {
            var rightValue = false;
            try{
                alert(rightValue || falseValue);
            }catch (e){
                alert(e);
            }
        }
3、逻辑与操作（短路操作）： 
        function andRight() {
            var rightValue = false;
            alert(rightValue && falseValue);
        }
4、逻辑与操作（报错）： 
        function andFalse() {
            var rightValue = true;
            try{
                alert(rightValue && falseValue);
            }catch (e){
                alert(e);
            }
        }
